```
#!/usr/bin/env bash
# Purpose: probe nvidia-smi for free VRAM; warn if any GPU < THRESHOLD_GB.
# Usage: ./check_gpu_mem.sh [threshold_gb=16]
set -euo pipefail

THRESHOLD_GB=${1:-16}  # Default 16 GB — covers 7B QLoRA training + headroom.

if ! command -v nvidia-smi >/dev/null 2>&1; then
  printf "error: nvidia-smi not found (no NVIDIA driver?)\n" >&2
  exit 2
fi

# Use --query-gpu instead of parsing nvidia-smi text — stable across driver versions.
mapfile -t free_mib < <(nvidia-smi --query-gpu=memory.free --format=csv,noheader,nounits)

ok=0
warned=0
for i in "${!free_mib[@]}"; do
  free_gb=$(( free_mib[i] / 1024 ))
  if (( free_gb < THRESHOLD_GB )); then
    printf "GPU %d: %d GB free (below %d GB threshold)\n" "$i" "$free_gb" "$THRESHOLD_GB" >&2
    warned=1
  else
    printf "GPU %d: %d GB free — ok\n" "$i" "$free_gb"
    ok=$(( ok + 1 ))
  fi
done

exit $warned
```
